int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        int arr[n];
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        vector<int> ans = getStarAndSuperStar(arr, n);
        for (int i=1;i<ans.size();i++) {
            cout << ans[i] << " ";
        }
        cout << "\n";
        cout << ans[0] << "\n";
    }
    return 0;
}




vector<int> getStarAndSuperStar(int arr[], int n) {
      vector<int> ans;
      int max = INT_MIN;
      for(int i=n-1;i>=0;i--)
      {
          if(arr[i]>max)
          {
              ans.push_back(arr[i]);
              max=arr[i];
          }
          
      }
     
     int count=0;
     for(int i=0;i<n;i++)
     {
         if(arr[i]==max)
         {
             count++;
         }
     }
     
     if(count>1)
     ans.push_back(-1);
     else
     ans.push_back(max);
     
      reverse(ans.begin(),ans.end());
      return ans;
      
      
}
